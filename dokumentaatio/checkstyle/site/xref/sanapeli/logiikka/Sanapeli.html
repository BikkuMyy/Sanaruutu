1   package sanapeli.logiikka;
2   import java.util.ArrayList;
3   import javax.swing.SwingUtilities;
4   import sanapeli.kayttoliittyma.Kayttoliittyma;
5   
6   /**
7    * Luokka vastaa Pelialustan ja Kayttoliittyman luomisesta sekä pelin
8    * käynnistämisestä.
9    */
10  
11  public class Sanapeli implements Peliruudukko {
12  
13      private Pelialusta pelialusta;
14      private SananTarkistaja tarkistaja;
15  
16      /**
17       * Konstruktori.
18       */
19      public Sanapeli() {
20          this.pelialusta = new Pelialusta();
21          this.tarkistaja  = new SananTarkistaja("sanalista.txt");
22      }
23  
24      /**
25       * Metodi luo ohjelman käyttöliittymän ja pelialustan alkutilanteen.
26       */
27      public void kaynnista() {
28          Kayttoliittyma kayttoliittyma = new Kayttoliittyma(this);
29          SwingUtilities.invokeLater(kayttoliittyma);
30          pelialusta.luoAlkuTilanne();
31      }
32  
33      @Override
34      public Ruutu[][] getRuudukko() {
35          return pelialusta.getRuudukko();
36      }
37  
38      @Override
39      public Ruutu getRuutu(int y, int x) {
40          return pelialusta.getRuutu(y, x);
41      }
42  
43      @Override
44      public ArrayList<Ruutu> getValitut() {
45          return pelialusta.getValitut();
46      }
47  
48      @Override
49      public boolean tarkistaSana(String sana) {
50          return tarkistaja.tarkistaSana(sana);
51      }
52  
53      @Override
54      public void paivitaRuudukko() {
55          pelialusta.paivitaRuudukko();
56      }
57  }
